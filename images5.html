Complementary Colors

createGraphics two Canvases in One

April 02, 2020 David Nieto

Los siguientes ejemplos tienen el propósito de practicar la transición entre el lenguaje Processing y el lenguaje p5.js por medio del uso de diferentes funciones y ejemplos relacionados con el procesamiento de imágenes. Adicionalmente se hará uso de la función createGraphics() como se vio en el ejemplo DRAW AND PAINT con el objetivo de crear dos lienzos en la ventana principal, para poder observar y comparar los cambios de la imagen obtenida con la imagen original.

El siguiente programa está basado en los ejemplos trabajados durante la clase de Computación Visual y el estudio de la documentación Complementary colors. Este ejemplo tiene como objetivo calcular y observar los colores complementarios u opuestos de los pixeles de la imagen en el modelo RGB.


El cálculo de los colores opuestos o complementarios es muy sencillo, pero con ello se logra obtener un efecto muy interesante al conseguir como resultado una imagen invertida o el negativo de la imagen. Si sabemos que el rango de valor de un pixel va de 0 a 255, como se vio en el ejemplo FIRST STEPS, es posible calcular su valor opuesto igualando el valor de cada componente RGB del pixel a 255 y sustrayendo el componente RGB del pixel original; es decir, 255 – Rojo = Cian, 255 - Verde = Magenta y 255 - Azul = Amarillo.

p5*js Code

var lienzo_01; // Lienzo imagen original
var lienzo_02; // Lienzo imagen invertida
var img_01;
var img_02;

let c =false;
let m =false;
let y =false;

function setup() {
	createCanvas(772, 600);
	img_01 = loadImage('https://upload.wikimedia.org/wikipedia/commons/thumb/c/c0/Ara_macao_qtl1.jpg/386px-Ara_macao_qtl1.jpg');
	img_02 = loadImage('https://upload.wikimedia.org/wikipedia/commons/thumb/c/c0/Ara_macao_qtl1.jpg/386px-Ara_macao_qtl1.jpg');
	lienzo_01 = createGraphics(386, 600);
	lienzo_02 = createGraphics(386, 600);
}

function draw() {
	drawImage_01();
	drawImage_02();

	image(lienzo_01, 0, 0);
	image(lienzo_02, 386, 0);
}
// Dibuja la imagen de la Izquierda
function drawImage_01() {
	lienzo_01.image(img_01, 0, 0);
	var title = "IMAGEN ORIGINAL";
	
	lienzo_01.textSize(14);
	lienzo_01.textAlign(CENTER);
	lienzo_01.text(title, 0, 20,lienzo_01.width); 
}
// Dibuja la imagen de la Derecha
function drawImage_02() {
	var title = "IMAGEN OPUESTA";
	
	lienzo_02.textSize(14);
	lienzo_02.textAlign(CENTER);
	lienzo_02.text(title, 0, 20,lienzo_02.width);
	
	loadPixels();
	img_02.loadPixels();
	let img = createImage(386, 600);
	img.loadPixels();
	
	for (let index = 0; index < 4 * (width * height); index += 4){
		let r=img_02.pixels[index+0]; // Componente Red
		let g=img_02.pixels[index+1]; // Componente Green
		let b=img_02.pixels[index+2]; // Componente Blue
		let a=img_02.pixels[index+3]; // Componente Alpha

		if (c){
			img.pixels[index+0]=255-r; // Se calcula el opuesto del Rojo (cian)
		}else{
			img.pixels[index+0]=r;
		}
		if (m){
			img.pixels[index+1]=255-g; // Se calcula el opuesto del Verde (magenta)
		}else{
			img.pixels[index+1]=g;
		}
		if (y){
			img.pixels[index+2]=255-b; // Se calcula el opuesto del Azul (amarrillo)
		}else{
			img.pixels[index+2]=b;
		}		
		img.pixels[index+3]=a;
	}
	img.updatePixels();
	image(img, 386, 0);
}
// Se ejecuta cuando se presiona cualquier tecla
function keyPressed() {
	if (key === '1') { // Colores complementarios
		c=true;
		m=true;
		y=true;
	} else if (key === '0') { // Imagen original
		c=false;
		m=false;
		y=false;
	} else if (key === 'r' || key === 'R') { // Complemento Rojo
		c=true;
	} else if (key === 'g' || key === 'G') { // Complemento Verde
		m=true;
	} else if (key === 'b' || key === 'B') { // Complemento Azul
		y=true;
	}
}
